\newcommand{\calculateSavingThrow}[3]{%
	\ifthenelse{\equal{#3}{P}}%
	{%
		\ifnum \intcalcAdd{#1}{\intcalcAdd{#2}{\ProficiencyValue}}>0%
			+%
		\fi%
		\intcalcAdd{#1}{\intcalcAdd{#2}{\ProficiencyValue}}%
	}%
	{%
		\ifthenelse{\equal{#3}{E}}%
		{%
			\ifnum \intcalcAdd{#1}{\intcalcAdd{#2}{\intcalcMul{\ProficiencyValue}{2}}}>0%
				+%
			\fi%
			\intcalcAdd{#1}{\intcalcAdd{#2}{\intcalcMul{\ProficiencyValue}{2}}}%
		}%
		{%
			\ifthenelse{\equal{\JackOfAllTradesValue}{1}}%
			{%
				\ifnum\intcalcAdd{#1}{ \intcalcAdd{#2}{\intcalcShr{\ProficiencyValue}}}>0%
					+%
				\fi%
				\intcalcAdd{#1}{\intcalcAdd{#2}{\intcalcShr{\ProficiencyValue}}}%
			}%
			{%
				\ifnum \intcalcAdd{#1}{#2}>0%
					+%
				\fi%
				\intcalcAdd{#1}{#2}
			}%
		}%
	}%
}%

\newcommand{\setAttackBonusValues}[3]{%
	\ifthenelse{\equal{#1}{STR}}{%
		\renewcommand{\tempAbilityScoreValue}{\StrengthScoreValue}
	}{%
		\ifthenelse{\equal{#1}{DEX}}{%
			\renewcommand{\tempAbilityScoreValue}{\DexterityScoreValue}
		}{%
			\ifthenelse{\equal{#1}{CHA}}{%
				\renewcommand{\tempAbilityScoreValue}{\CharismaScoreValue}
			}{%
				\ifthenelse{\equal{#1}{INT}}{%
					\renewcommand{\tempAbilityScoreValue}{\IntelligenceScoreValue}
				}{%
				}%
			}%
		}%
	}%
	
	\ifthenelse{\equal{#2}{P}}{%
		\renewcommand{\tempAttackProficiencyModifier}{\ProficiencyValue}%
	}{%
		\ifthenelse{\equal{#2}{E}}{%
			\renewcommand{\tempAttackProficiencyModifier}{\intcalcMul{2}{\ProficiencyValue}}%
		}{%
			0%
		}%
	}%
	
	\renewcommand{\tempAttackBonusModifiers}{#3}%
}%

		}{%
			\intcalcAdd{%
				\tempAttackProficiencyModifier%
			}{%
				\tempAttackBonusModifiers%
			}%
		}%
	}
}%
